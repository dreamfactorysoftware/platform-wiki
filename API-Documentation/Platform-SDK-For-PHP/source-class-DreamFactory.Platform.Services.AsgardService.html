 85:     /**
 86:      * Gather a set of stats for a local DSP
 87:      *
 88:      * @return null|bool
 89:      */
 90:     public static function getStats()
 91:     {
 92:         $_stats = new self();
 93: 
 94:         //  Connect to the user DSP db
 95:         $_db = Pii::db();
 96:         $_pdo = $_db->getPdoInstance();
 97:         $_dspName = Pii::getParam( 'dsp.name' );
 98: 
 99:         //  Get the stats for this DSP
100:         $_stats->set( 'database', $_dbStats = static::_databaseStats( $_dspName, $_db, 'dreamfactory' ) );
101:         $_stats->set( 'state', $_dbStats['state'] );
102: 
103:         //  Unactivated DSPs don't have these tables
104:         if ( 'active' == $_dbStats['state'] )
105:         {
106:             $_stats->set( 'apps', static::_appStats( $_dspName, $_pdo ) );
107: 
108:             $_stats->set(
109:                 'app_groups',
110:                 array(
111:                     'counts' => static::_rowCount( 'df_sys_app_group', $_dspName, $_pdo )
112:                 )
113:             );
114: 
115:             $_stats->set(
116:                 'roles',
117:                 array(
118:                     'counts' => static::_rowCount( 'df_sys_role', $_dspName, $_pdo )
119:                 )
120:             );
121: 
122:             $_stats->set(
123:                 'services',
124:                 array(
125:                     'counts' => static::_rowCount( 'df_sys_service', $_dspName, $_pdo )
126:                 )
127:             );
128: 
129:             $_stats->set(
130:                 'users',
131:                 array(
132:                     'counts' => static::_rowCount( 'df_sys_user', $_dspName, $_pdo )
133:                 )
134:             );
135:         }
136: 
137:         //  Log it to the database...
138:         Stat::create( Stat::TYPE_ASGARD, Pii::user()->getId(), $_stats->contents() );
139: 
140:         return $_stats->contents();
141:     }

143:     /**
144:      * @param string         $dspName
145:      * @param \CDbConnection $db
146:      * @param string         $dbName The name of the database
147:      *
148:      * @return bool
149:      */
150:     protected static function _databaseStats( $dspName, $db, $dbName )
151:     {
152:         $_details = array();
153:         $_tableCounter = new Counter( 'system' );
154:         $_rowCounter = new Counter( 'system' );
155:         $_len = strlen( static::SYSTEM_TABLE_NAME_PREFIX );
156:         $_pdo = $db->getPdoInstance();
157: 
158:         $_schema = $db->getSchema();
159:         $_tables = $_schema->getTableNames();
160: 
161:         if ( empty( $_tables ) || 1 == sizeof( $_tables ) )
162:         {
163:             //  Only a cache table means unactivated DSP
164:             if ( 1 == sizeof( $_tables ) )
165:             {
166:                 $_tableCounter->increment( 'system' );
167:             }
168: 
169:             return array(
170:                 'table_counts' => $_tableCounter->getCounters(),
171:                 'row_counts'   => $_rowCounter->getCounters(),
172:                 'details'      => $_details,
173:                 'state'        => 'inactive',
174:             );
175:         }
176: 
177:         /**
178:          * Table stats
179:          */
180: 
181:         /** @var $_tables \CDbTableSchema[] */
182:         foreach ( $_tables as $_table )
183:         {
184:             $_system = false;
185: 
186:             if ( static::SYSTEM_TABLE_NAME_PREFIX == substr( $_table, 0, $_len ) )
187:             {
188:                 $_system = true;
189:             }
190: 
191:             $_tableCounter->increment( $_system ? '*' : null );
192:             $_rowCounter->increment( $_system ? '*' : null, static::_rowCount( $_table, $dspName, $_pdo ) );
193: 
194:             unset( $_tableSchema, $_tableName );
195:         }
196: 
197:         unset( $_tables );
198: 
199:         /**
200:          * DB Stats
201:          */
202:         $_sql = <<<MYSQL
203: SELECT
204:     SUM( data_length + index_length ) / 1024 / 1024 AS mb_used_nbr,
205:     SUM( data_free ) / 1024 / 1024 AS mb_free_nbr
206: FROM
207:     information_schema.TABLES
208: WHERE
209:     table_schema = :table_schema
210: GROUP BY
211:     table_schema
212: MYSQL;
213: 
214:         if ( false !== ( $_row = Sql::find( $_sql, array( ':table_schema' => $dbName ), Pii::pdo() ) ) )
215:         {
216:             $_details['disk_space_used'] = $_row['mb_used_nbr'];
217:             $_details['disk_space_free'] = $_row['mb_free_nbr'];
218:         }
219: 
220:         return array(
221:             'table_counts' => $_tableCounter->getCounters(),
222:             'row_counts'   => $_rowCounter->getCounters(),
223:             'size_counts'  => $_details,
224:             'state'        => 'active',
225:         );
226:     }

228:     /**
229:      * @param string $dspName
230:      * @param \PDO   $pdo
231:      *
232:      * @return bool
233:      */
234:     protected static function _appStats( $dspName, $pdo )
235:     {
236:         Sql::setConnection( $pdo );
237: 
238:         $_sql = <<<MYSQL
239: SELECT
240:     a.api_name,
241:     a.is_active,
242:     a.url,
243:     a.is_url_external,
244:     a.import_url
245: FROM
246:     df_sys_app a
247: ORDER BY
248:     a.api_name
249: MYSQL;
250: 
251:         if ( false === ( $_reader = Sql::query( $_sql ) ) )
252:         {
253:             Log::error( 'Error retrieving app information from DSP "' . $dspName . '":' . print_r( $pdo->errorInfo(), true ) );
254: 
255:             return false;
256:         }
257: 
258:         $_details = array();
259:         $_counter = new Counter( 'system' );
260: 
261:         foreach ( $_reader as $_row )
262:         {
263:             $_system = false;
264: 
265:             foreach ( static::$_repoUrls as $_repo )
266:             {
267:                 if ( false !== strpos( $_row['import_url'], $_repo ) )
268:                 {
269:                     $_system = true;
270:                     break;
271:                 }
272: 
273:                 unset( $_repo );
274:             }
275: 
276:             $_details[$_row['api_name']] = array(
277:                 'apiName'      => $_row['api_name'],
278:                 'active'       => $_row['is_active'] == 1 ? true : false,
279:                 'url'          => $_row['url'],
280:                 'external_url' => $_row['is_url_external'] == 1 ? true : false,
281:                 'import_url'   => $_row['import_url'],
282:                 'system'       => $_system,
283:             );
284: 
285:             $_counter->increment( $_system ? '*' : null );
286: 
287:             unset( $_row );
288:         }
289: 
290:         unset( $_reader );
291: 
292:         return array(
293:             'counts'  => $_counter->getCounters(),
294:             'details' => $_details,
295:         );
296:     }
