263:     public static function replaceLookupsInStrings( &$string )
264:     {
265:         if ( false !== strpos( $string, '{' ) )
266:         {
267:             $_change = false;
268:             // brute force, yeah this could be better
269:             $_exploded = explode( ' ', $string );
270:             foreach ( $_exploded as $_key => $_word )
271:             {
272:                 if ( is_string( $_word ) )
273:                 {
274:                     $_end = strlen( $_word ) - 1;
275:                     if ( ( 0 === strpos( $_word, '{' ) ) && ( $_end === strrpos( $_word, '}' ) ) )
276:                     {
277:                         if ( Session::getLookupValue( substr( $_word, 1, $_end - 1 ), $_value ) )
278:                         {
279:                             $_change = true;
280:                             $_exploded[$_key] = $_value;
281:                         }
282:                     }
283:                 }
284:             }
285: 
286:             if ( $_change )
