 84:     /**
 85:      * @throws \InvalidArgumentException
 86:      * @throws \DreamFactory\Platform\Exceptions\BadRequestException
 87:      * @throws \DreamFactory\Platform\Exceptions\MisconfigurationException
 88:      * @throws \Exception
 89:      * @return array|bool
 90:      */
 91:     protected function _handleResource()
 92:     {
 93:         if ( empty( $this->_resource ) )
 94:         {
 95:             if ( static::GET == $this->_action )
 96:             {
 97:                 return $this->_listResources();
 98:             }
 99: 
100:             return false;
101:         }
102: 
103:         switch ( $this->_resource )
104:         {
105:             case 'session':
106:                 //  Handle remote login
107:                 if ( HttpMethod::POST == $this->_action && Pii::getParam( 'dsp.allow_remote_logins' ) )
108:                 {
109:                     $_provider = FilterInput::post( 'provider', null, FILTER_SANITIZE_STRING );
110: 
111:                     if ( !empty( $_provider ) )
112:                     {
113:                         Pii::redirect( '/web/remoteLogin?pid=' . $_provider . '&flow=' . Flows::SERVER_SIDE );
114:                     }
115:                 }
116: 
117:                 $obj = new Session( $this );
118:                 $result = $obj->processRequest( null, $this->_action );
119:                 break;
120: 
121:             case 'custom':
122:                 $obj = new CustomSettings( $this, $this->_resourceArray );
123:                 $result = $obj->processRequest( null, $this->_action );
124:                 break;
125: 
126:             case 'device':
127:                 $obj = new Device( $this, $this->_resourceArray );
128:                 $result = $obj->processRequest( null, $this->_action );
129:                 break;
130: 
131:             case 'profile':
132:                 $obj = new Profile( $this );
133:                 $result = $obj->processRequest( null, $this->_action );
134:                 break;
135: 
136:             case 'challenge': // backward compatibility
137:             case 'password':
138:                 $obj = new Password( $this );
139:                 $result = $obj->processRequest( null, $this->_action );
140:                 break;
141: 
142:             case 'confirm': // backward compatibility
143:             case 'register':
144:                 $obj = new Register( $this );
145:                 $result = $obj->processRequest( null, $this->_action );
146:                 break;
147: 
148:             case 'ticket':
149:                 switch ( $this->_action )
150:                 {
151:                     case static::GET:
152:                         $result = $this->userTicket();
153:                         break;
154:                     default:
155:                         return false;
156:                 }
157:                 break;
158: 
159:             default:
160:                 return false;
161:                 break;
162:         }
163: 
164:         //  Send out an event
165:         return $result;
166:     }

170:     /**
171:      * userTicket generates a SSO timed ticket for current valid session
172:      *
173:      * @return array
174:      * @throws \Exception
175:      */
176:     public static function userTicket()
177:     {
178:         try
179:         {
180:             $userId = Session::validateSession();
181:         }
182:         catch ( \Exception $ex )
183:         {
184:             Session::userLogout();
185:             throw $ex;
186:         }
187: 
188:         // regenerate new timed ticket
189:         $timestamp = time();
190:         $ticket = Utilities::encryptCreds( $userId . ',' . $timestamp, 'gorilla' );
191: 
192:         return array( 'ticket' => $ticket, 'ticket_expiry' => time() + ( 5 * 60 ) );
193:     }
