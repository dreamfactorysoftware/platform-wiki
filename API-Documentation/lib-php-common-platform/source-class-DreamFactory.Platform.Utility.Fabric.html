128:     /**
129:      * @throws \CHttpException
130:      * @return array|mixed
131:      * @throws RuntimeException
132:      * @throws CHttpException
133:      */
134:     public static function initialize()
135:     {
136:         global $_dbName, $_instance, $_dspName;
137: 
138:         //  If this isn't a cloud request, bail
139:         $_host = static::getHostName();
140: 
141:         if ( !static::hostedPrivatePlatform() && false === strpos( $_host, static::DSP_DEFAULT_SUBDOMAIN ) )
142:         {
143:             static::_errorLog( 'Attempt to access system from non-provisioned host: ' . $_host );
144:             throw new \CHttpException( HttpResponse::Forbidden, 'You are not authorized to access this system you cheeky devil you. (' . $_host . ').' );
145:         }
146: 
147:         //  What has it gots in its pocketses? Cookies first, then session
148:         $_privateKey = FilterInput::cookie( static::PrivateFigNewton, FilterInput::session( static::PrivateFigNewton ), \Kisma::get( 'platform.user_key' ) );
149:         $_dspName = str_ireplace( static::DSP_DEFAULT_SUBDOMAIN, null, $_host );
150: 
151:         $_dbConfigFileName = str_ireplace(
152:             '%%INSTANCE_NAME%%',
153:             $_dspName,
154:             static::DSP_DB_CONFIG_FILE_NAME_PATTERN
155:         );
156: 
157:         //  Try and get them from server...
158:         if ( false === ( list( $_settings, $_instance ) = static::_checkCache( $_host ) ) )
159:         {
160:             //  Otherwise we need to build it.
161:             $_parts = explode( '.', $_host );
162:             $_dbName = str_replace( '-', '_', $_dspName = $_parts[0] );
163: 
164:             //  Otherwise, get the credentials from the auth server...
165:             $_response = Curl::get( static::DEFAULT_AUTH_ENDPOINT . '/' . $_dspName . '/database' );
166: 
167:             if ( HttpResponse::NotFound == Curl::getLastHttpCode() )
168:             {
169:                 static::_errorLog( 'DB Credential pull failure. Redirecting to df.com: ' . $_host );
170:                 header( 'Location: https://www.dreamfactory.com/dsp-not-found?dn=' . urlencode( $_dspName ) );
171:                 exit();
172:             }
173: 
174:             if ( is_object(
175:                      $_response
176:                  ) && isset( $_response->details, $_response->details->code ) && HttpResponse::NotFound == $_response->details->code
177:             )
178:             {
179:                 static::_errorLog( 'Instance "' . $_dspName . '" not found during web initialize.' );
180:                 throw new \CHttpException( HttpResponse::NotFound, 'Instance not available.' );
181:             }
182: 
183:             if ( !$_response || !is_object( $_response ) || false == $_response->success )
184:             {
185:                 static::_errorLog( 'Error connecting to authentication service: ' . print_r( $_response, true ) );
186:                 throw new \CHttpException( HttpResponse::InternalServerError, 'Cannot connect to authentication service' );
187:             }
188: 
189:             $_instance = $_cache = $_response->details;
190:             $_dbName = $_instance->db_name;
191:             $_dspName = $_instance->instance->instance_name_text;

193:             $_privatePath = $_cache->private_path;
194:             $_privateKey = basename( dirname( $_privatePath ) );

196:             //  Stick this in persistent storage
197:             \Kisma::set(
198:                 array(
199:                     'dsp.credentials'              => $_cache,
200:                     'platform.dsp_name'            => $_dspName,
201:                     'platform.private_path'        => $_privatePath,
202:                     'platform.storage_key'         => $_instance->storage_key,
203:                     'platform.private_storage_key' => $_privateKey,
204:                     'platform.db_config_file'      => $_privatePath . '/' . $_dbConfigFileName,

206:                 )

209:             /** @noinspection PhpIncludeInspection */
210:             //  File should be there from provisioning... If not, tenemos una problema!
211:             $_settings = require( $_privatePath . '/' . $_dbConfigFileName );

213:             if ( !empty( $_settings ) )

231:     /**
232:      * @param string $host
233:      *
234:      * @return bool|mixed
235:      */
236:     protected static function _checkCache( $host )
237:     {
238:         //  See if file is available and return it, or expire it...
239:         if ( file_exists( $_cacheFile = static::_cacheFileName( $host ) ) )
240:         {
241:             //  No session or expired?
242:             if ( Pii::isEmpty( session_id() ) || ( time() - fileatime( $_cacheFile ) ) > static::EXPIRATION_THRESHOLD )
243:             {
244:                 @unlink( $_cacheFile );
245: 
246:                 return false;
247:             }
248: 
249:             if ( false !== ( $_data = json_decode( file_get_contents( $_cacheFile ), true ) ) )
250:             {
251:                 return array( $_data['settings'], $_data['instance'] );
252:             }
253:         }
254: 
255:         return false;
256:     }

294:      * @param string $id
295:      *
296:      * @return array
297:      */
298:     public static function getProviderCredentials( $id = null )
299:     {
300:         if ( !static::fabricHosted() && !static::hostedPrivatePlatform() )
301:         {
302:             Log::info( 'Global provider credential pull skipped: not hosted entity.' );
303: 
304:             return array();
305:         }
306: 
307:         //  Otherwise, get the credentials from the auth server...
308:         $_url = static::DEFAULT_PROVIDER_ENDPOINT . '/';
309: 
310:         if ( null !== $id )
311:         {
312:             $_url .= $id . '/';
313:         }
314: 
315:         $_response = Curl::get( $_url . '?oasys=' . urlencode( Pii::getParam( 'oauth.salt' ) ) );
316: 
317:         if ( HttpResponse::Ok != Curl::getLastHttpCode() || !$_response->success )
318:         {
319:             static::_errorLog( 'Global provider credential pull failed: ' . Curl::getLastHttpCode() . PHP_EOL . print_r( $_response, true ) );
320: 
321:             return array();
322:         }
323: 
324:         return Option::get( $_response, 'details', array() );
325:     }
