 65:     /**
 66:      * Constructs a virtual platform path
 67:      *
 68:      * @param string $type            The type of path, used as a key into config
 69:      * @param string $append
 70:      * @param bool   $createIfMissing If true and final directory does not exist, it is created.
 71:      *
 72:      * @param bool   $includesFile
 73:      *
 74:      * @throws \InvalidArgumentException
 75:      * @throws \Kisma\Core\Exceptions\FileSystemException
 76:      * @return string
 77:      */
 78:     protected static function _getPlatformPath( $type, $append = null, $createIfMissing = true, $includesFile = false )
 79:     {
 80:         static $_cache = array();
 81: 
 82:         $_appendage = ( $append ? '/' . ltrim( $append, '/' ) : null );
 83: 
 84:         if ( !LocalStorageTypes::contains( $_tag = Inflector::neutralize( $type ) ) )
 85:         {
 86:             throw new \InvalidArgumentException( 'Type "' . $type . '" is invalid.' );
 87:         }
 88: 
 89:         //  Make a cache tag that includes the requested path...
 90:         $_cacheTag = $_tag . '/' . Inflector::neutralize( $_appendage );
 91: 
 92:         if ( null === ( $_path = Option::get( $_cache, $_cacheTag ) ) )
 93:         {
 94:             $_path = trim( Pii::getParam( $_tag ) );
 95: 
 96:             if ( empty( $_path ) )
 97:             {
 98:                 $_path = \Kisma::get( 'app.project_root' ) . '/storage';
 99:                 Log::notice( 'Empty path for platform path type "' . $type . '". Defaulting to "' . $_path . '"' );
100:             }
101: 
102:             $_checkPath = $_path . $_appendage;
103: 
104:             if ( $includesFile )
105:             {
106:                 $_checkPath = dirname( $_checkPath );
107:             }
108: 
109:             if ( true === $createIfMissing && !is_dir( $_checkPath ) )
110:             {
111:                 if ( false === @\mkdir( $_checkPath, 0777, true ) )
112:                 {
113:                     throw new FileSystemException( 'File system error creating directory: ' . $_checkPath );
114:                 }
115:             }
116: 
117:             $_path .= $_appendage;
118: 
119:             //  Store path for next time...
120:             Option::set( $_cache, $_cacheTag, $_path );
121:         }
122: 
123:         return $_path;
124:     }

260:     /**
261:      * @param string $namespace
262:      *
263:      * @return string
264:      */
265:     public static function uuid( $namespace = null )
266:     {
267:         static $_uuid = null;
268: 
269:         $_hash = strtoupper(
270:             hash(
271:                 'ripemd128',
272:                 uniqid( '', true ) . ( $_uuid ? : microtime( true ) ) . md5(
273:                     $namespace . $_SERVER['REQUEST_TIME'] . $_SERVER['HTTP_USER_AGENT'] . $_SERVER['LOCAL_ADDR'] . $_SERVER['LOCAL_PORT'] .
274:                     $_SERVER['REMOTE_ADDR'] . $_SERVER['REMOTE_PORT']
275:                 )
276:             )
277:         );
278: 
279:         $_uuid =
280:             '{' .
281:             substr( $_hash, 0, 8 ) .
282:             '-' .
283:             substr( $_hash, 8, 4 ) .
284:             '-' .
285:             substr( $_hash, 12, 4 ) .
286:             '-' .
287:             substr( $_hash, 16, 4 ) .
288:             '-' .
289:             substr( $_hash, 20, 12 ) .
290:             '}';
291: 
292:         return $_uuid;
293:     }
